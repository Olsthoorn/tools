\begin{alltt}
\textcolor{comment}{% Large-diameter well drawdown with well storage (Bouton, 1963)}

\textcolor{comment}{%\% Specify well}
rW  = 5;
zW  =-5;
k   = 1;
ss  = 0.000;
Sy  = 0.05;
Qw  =-130;

\textcolor{comment}{%\% Grid}
rGr = [logspace(0,4,50), rW+[-0.5 -0.25 -0.1 0 0.1 0.25 0.5 1]];
yGr = [0:-1:-20, zW+[-0.5 -0.25 -0.1 0 0.1 0.25 0.5]];
t   = logspace(0,3,31);

gr = grid2DObj(rGr,yGr,\textcolor{string}{'AXIAL'},true);

\textcolor{comment}{%\% Define inWell domain}
inWell = gr.Ym>zW \& gr.Rm<rW;

\textcolor{comment}{%\% Arrays}
IBOUND = gr.const(1);
K  = gr.const(k); K(inWell)=10000;
Ss            = gr.const(ss);  \textcolor{comment}{% elastic storage }
Ss(1,:)       = Sy/gr.dy(1);   \textcolor{comment}{% specific yield}
Ss(1,gr.rm<rW)= 1/gr.dy(1);    \textcolor{comment}{% within well Ss=1 (free surface)}

IH = gr.const(0);
FQ = gr.const(0); FQ(1,2)=Qw;

\textcolor{comment}{%\% Run model}
[Phi,Qt,Qr,Qz,Qs]=fdm2t(gr,t,K,K,Ss,IBOUND,IH,FQ);

\textcolor{comment}{%\% Analytic solution Theis}
kD = sum(K( :,end) .* gr.dy);
SY = sum(Ss(:,end) .* gr.dy);
fi = Qw/(4*pi*kD)*expint(bsxfun(@rdivide,gr.rm.\^{}2*SY,4*kD*t(:)));

\textcolor{comment}{%\% Visualize}
close all
figure; axes(\textcolor{string}{'nextPlot'},\textcolor{string}{'add'},\textcolor{string}{'xScale'},\textcolor{string}{'log'});
title(\textcolor{string}{'Drawdown insize pp large diameter well, compare with Theis'});
xlabel(\textcolor{string}{'r [m]'}); ylabel(\textcolor{string}{'dd [m]'});
plot(gr.rm(:),fi,\textcolor{string}{'k'});
plot(gr.rm,squeeze(Phi(1,:,:)),\textcolor{string}{'+'});
\end{alltt}
