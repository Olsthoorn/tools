\begin{alltt}
\textcolor{comment}{%\% modelScript -- transient model for Theis well case}
\textcolor{comment}{% Transient axially symmetric model simulating transient extraction}
\textcolor{comment}{% from a confined aquifer according to Theis. Comparison with anlytic}
\textcolor{comment}{% solution.}

close all;

\textcolor{comment}{%\% Data for theis well and aquifer}
Qw =-2400;
rW = 0.1;
kD = 200; D=20; k = kD/D;
S  = 0.1;       Ss = S/D;

t=logspace(-3,2,51);

\textcolor{comment}{%\% grid}
rGr = logspace(log10(rW),4,41);
yGr = [0  -20];
gr = grid2DObj(rGr,yGr,\textcolor{string}{'AXIAL'});

IBOUND = gr.const(1);

HI = gr.const(0);
FQ = gr.const(0);  FQ(1,1) = Qw;

TheisR = @(t) Qw/(4*pi*kD) * expint(gr.rm.\^{}2*S./(4*kD*t));
TheisT = @(r) Qw/(4*pi*kD) * expint(   r .\^{}2*S./(4*kD*t));

[Phi,Qin,Qs]=fdm2t(gr,t,k,k,Ss,IBOUND,HI,FQ);

[Phi1,Qin1,Qx1,Qy1,Qs1]=fdm2ct(gr,t,k,k,Ss,IBOUND,HI,FQ,\textcolor{string}{'axial'});

\textcolor{comment}{%\% Drawdown as function of distance}
close all
figure;

subplot(2,1,1,\textcolor{string}{'nextplot'},\textcolor{string}{'add'},\textcolor{string}{'xScale'},\textcolor{string}{'log'});
title( \textcolor{string}{'Theis drawdown as function of time for different r'});
xlabel(\textcolor{string}{'t [d]'}); ylabel(\textcolor{string}{'dd [m]'});

\textcolor{keyword}{for} ir=1:5:gr.Nr
   plot(t,squeeze(Phi(1,ir,:)),\textcolor{string}{'bo'}); \textcolor{comment}{% numerical }
   plot(t,TheisT(gr.rm(ir)),\textcolor{string}{'k'}); \textcolor{comment}{% analytical}
\textcolor{keyword}{end}

subplot(2,1,2,\textcolor{string}{'nextplot'},\textcolor{string}{'add'},\textcolor{string}{'xScale'},\textcolor{string}{'log'});
title( \textcolor{string}{'Theis drawdown as function of r for different times'});
xlabel(\textcolor{string}{'r [m]'}); ylabel(\textcolor{string}{'dd [m]'});

\textcolor{keyword}{for} it=5:5:length(t)
   plot(gr.rm,Phi (1,:,it)  ,\textcolor{string}{'bo'}); \textcolor{comment}{% numerical}
   plot(gr.rm,Phi1(1,:,it)  ,\textcolor{string}{'rx'});
   plot(gr.rm,TheisR(t(it)), \textcolor{string}{'k'}); \textcolor{comment}{% analytical}
\textcolor{keyword}{end}
\end{alltt}
