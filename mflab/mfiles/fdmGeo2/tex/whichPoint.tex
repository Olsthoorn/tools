\textcolor{keyword}{function} [I,Ix,Iy,i] = whichPoint(gr,I)
\textcolor{comment}{% Get indexes of MOC point in grid}
\textcolor{comment}{% TO 140416}
\textcolor{comment}{% I= gr.Nod*(N-1)+gr.Ny*(ix-1)+iy}

\textcolor{keyword}{if} nargin<3, n=1; \textcolor{keyword}{end}

I = I-1; \textcolor{comment}{% Compute zero based, which is simpler than one-based:}

i  = floor(I/gr.Nod);

I  = I - i*gr.Nod;

Ix = floor(I/gr.Ny);

Iy = I -  Ix*gr.Ny;

\textcolor{comment}{%export one-based}
I = I+1;
i = i+1;
Ix= Ix+1;
Iy= Iy+1;

\textcolor{comment}{%gr.Nod * (i-1) + (Ix-1) * gr.Ny + Iy;}
